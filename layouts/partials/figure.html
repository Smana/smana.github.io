{{/* Project-level override of the theme partial. Upstream renders the original file at full resolution and only offers a WebP / AVIF if a sibling file with that extension already exists on disk. Nothing generates those, so a post card ended up downloading the full-size thumbnail (up to 4.8 MB, 2147px wide) into a ~200px slot. Here the image is resized and encoded to WebP at build time, with a resized fallback in the original format. Intrinsic width/height are always emitted so the browser reserves the box before the image lands (no layout shift). */}} {{- $file := .file -}} {{- $image := .image -}} {{- $type := .type -}} {{- $diskPath := .diskPath -}} {{- $webPath := .webPath -}} {{- $alt := .alt -}} {{- $cap := .cap -}} {{- $classes := .classes -}} {{- $bundle := .bundle -}} {{- $fileWeb := .file -}} {{- if eq $bundle true -}} {{ $fileWeb = path.Join $webPath $file }} {{ if in $classes "image_svg" }} {{ if strings.HasPrefix $file "http" }} {{ $fileWeb = $file }} {{- end -}} {{- end -}} {{- end -}} {{/* Formats Hugo can resize. SVG has no raster pipeline; GIF would lose animation. */}} {{- $processable := false -}} {{- with $image -}} {{- if and .MediaType (in (slice "png" "jpeg" "webp" "tiff" "bmp") .MediaType.SubType) -}} {{- $processable = true -}} {{- end -}} {{- end -}} {{/* Target width per usage, so a card doesn't pull a full-resolution asset. */}} {{- $maxWidth := 1200 -}} {{- $sizes := "(max-width: 600px) 100vw, min(1200px, 100vw)" -}} {{- if eq $type "thumbnail" -}} {{- $maxWidth = 300 -}} {{- $sizes = "300px" -}} {{- else if eq $type "featured" -}} {{- $maxWidth = 1600 -}} {{- $sizes = "(max-width: 600px) 100vw, min(1600px, 100vw)" -}} {{- end -}}
